Countdown fixed 1. Using closure. 
      function countdownFixed1(num) {
        for(var i = 0; i <= num; i+=1) {
          (function(i) { 
            setTimeout(function() {
              alert(num - i);
            }, i*1000);
          })(i);
        }
      }

      function countdownFixed2(num) {
        for(var i=0; i <= num; i+=1) {
          setTimeout(function(n) {
