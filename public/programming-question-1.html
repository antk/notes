Given an array of n positive and negative integers, move negative integers to the left and positive integers to the right. Do this in O(n) time and do it in memory (do not create a temporary array). For example:

Before: [1, -1, -2, -3, 4, 6, -4, -8, 9]
After: [-1, -2, -3, -4, -8, 1, 4, 6, 9] 
      function a(arr) {
        var swapPos = 0;
        for(var i = 0; i < arr.length; i++) {
          var currElem = arr[i];
          var swapElem = arr[swapPos];
          if(currElem < 0) {
            arr[i] = swapElem;
            arr[swapPos] = currElem;
            swapPos++;
          }
        }
        return arr.sort(function(a,b) { return a - b; });
      }
